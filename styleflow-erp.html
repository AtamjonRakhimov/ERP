GitHub Actions Workflow

# .github/workflows/deploy.yml
name: StyleFlow CI/CD Pipeline

on:
  push:
    branches: [main, develop]

jobs:
  build-and-deploy:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout Code
        uses: actions/checkout@v3

      - name: Setup Node.js
        uses: actions/setup-node@v3
        with: { node-version: '18' }

      - name: Install & Build
        run: |
          npm install
          npm run build

      - name: Run Tests
        run: npm test

      - name: Build Docker Image
        run: |
          docker build -t styleflow-erp:${{ github.sha }} .
          docker tag styleflow-erp:${{ github.sha }} \
            ${{ secrets.ECR_REGISTRY }}/styleflow-erp:latest

      - name: Push to AWS ECR
        run: docker push ${{ secrets.ECR_REGISTRY }}/styleflow-erp:latest

      - name: Deploy to AWS ECS
        run: |
          aws ecs update-service \
            --cluster styleflow-prod \
            --service styleflow-erp \
            --force-new-deployment
